\begin{algorithm}
\SetAlgoLined
\KwData{$f(x)$, $a$, $b$, $N$, $\delta$, $\epsilon$}
\KwResult{Mid point of bracketing interval}
$a_0 = a$, $b_0 = b$, $k=0$\;
\While{$k < N$}
{
$c_k = \half(a_k + b_k)$\;
\If{$|f(c_k)| < \delta$}
{
	Return $c_k$\;
}
\eIf{$\sign f(c_k) \ne \sign f(b_k)$}
{
	$a_{k+1} = c_k$\;
	$b_{k+1} = b_k$\;
}
{
	$a_{k+1} = a_k$\;
	$b_{k+1} = c_k$\;
}
$k = k + 1$\;
\If{$b_k - a_k < \epsilon$}
{
	Return $\half(a_k + b_k)$\;
}
}
\caption{Bisection method}
\label{alg:bis1}
\end{algorithm}

\begin{algorithm}
\SetAlgoLined
\KwData{$f(x)$, $a$, $b$, $N$, $\delta$, $\epsilon$}
\KwResult{Mid point of bracketing interval}
$k=0$\;
\While{$k < N$}
{
$c = \half(a + b)$\;
\If{$|f(c)| < \delta$}
{
	Return $c$\;
}
\eIf{$\sign f(c) \ne \sign f(b)$}
{
	$a = c$\;
}
{
	$b = c$\;
}
$k = k + 1$\;
\If{$b - a < \epsilon$}
{
	Return $\half(a + b)$\;
}
}
\caption{Bisection method}
\label{alg:bis2}
\end{algorithm}

\begin{algorithm}
\SetAlgoLined
\KwData{$f(x)$, $f'(x)$, $x$, $N$, $\delta$, $\epsilon$}
\KwResult{Root}
$k=0$, $f = f(x)$\;
\While{$k < N$}
{
	$df = f'(x)$\;
	$e = - f/df$\;
	$x = x + e$\;
	$f = f(x)$\;
	\If{$|e| < |x| \epsilon$ or $|f| < \delta$}
	{
		Return $x$\;
	}
	$k = k + 1$\;
}
\caption{Newton-Raphson method}
\label{alg:newton}
\end{algorithm}
